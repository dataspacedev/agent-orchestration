```go
func (r *AgentPolicyReconciler) SetupWithManager(mgr ctrl.Manager) error {
    return ctrl.NewControllerManagedBy(mgr).
        For(&agentsv1alpha1.ClusterAgentPolicy{}).
        Watches(
            &corev1.Namespace{},
            handler.EnqueueRequestsFromMapFunc(r.namespaceToPolicy),
        ).
        Complete(r)
}

// namespaceToPolicy maps a Namespace event to reconcile requests for ClusterAgentPolicy
// (there's only one ClusterAgentPolicy "default" or the first found — list all and enqueue each)
func (r *AgentPolicyReconciler) namespaceToPolicy(ctx context.Context, obj client.Object) []reconcile.Request {
    // List all ClusterAgentPolicy objects and return one request per
    var list agentsv1alpha1.ClusterAgentPolicyList
    if err := r.List(ctx, &list); err != nil {
        return nil
    }
    requests := make([]reconcile.Request, len(list.Items))
    for i, cap := range list.Items {
        requests[i] = reconcile.Request{NamespacedName: types.NamespacedName{Name: cap.Name}}
    }
    return requests
}
```
